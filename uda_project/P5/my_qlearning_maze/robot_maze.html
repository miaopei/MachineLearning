import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    if random.uniform(0, 1) > epsilon: # 
        action = random.choice(actions) # 
    else: 
        action = max([(value, key) for key, value in qline.items()])[1] #  Q 
    return action
